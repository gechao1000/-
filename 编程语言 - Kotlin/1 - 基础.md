#### 字符串

```

#### 空值处理

```
// 可以为null
var a:String? = null

// 空安全调用，不为空才调用，为空直接返回null
name?.length

// 默认值
name?.length?:-1

// 非空断言，一定不是null（开发中很少使用）
name!!.length 
```

#### 4. 流程控制

```
if (true) { ... } else { ... }

// 加强版switch
var y = when(num) {
    1 -> "一"
    2,3,4 -> "二"
    else -> num
}

when {
 a > b -> println("大于")
 a < b -> println("大于")
 else -> println(“等于")
}

// 循环
for (n in 1..10) { println(n); }

#### 区间，集合

```
// 区间
var nums = 1..10
[1,10] 1..10
[1,10) 1 until 10

// 区间api
nums.reversed() //反转
nums.count() 

// List集合
不可变 listOf("a", "b", "c")
可变 mutableListOf(1,2,3);

// 遍历List
for (item in list) {
    println(item)
}
for ((index,item) in list.withIndex()) {
    println("${index} : ${item}")
}

// Map集合
不可变 mapOf("name" to "xiaomi", "age" to 12)
可变 mutableMapOf<String,String>()

// 遍历Map
for (entry in map) {
    println("${entry.key}: ${entry.value}")
}
```

#### 函数

```kotlin
// (参数列表):返回值
fun add(a:Int,b:Int):Int {
	return a + b;
}

// 单表达式
fun add(a:Int,b:Int):Int = a+b

// 具名参数，调用时指定参数名
show(name = "dd")

// 允许为null，默认值
fun show(name:String?, age:Int = 12) {...}

// 可变参数，最后一位
fun show(name:String, vararg scores:Int) { score.size }

// 尾递归，避免栈溢出
tailrec fun sum(num:Int, total:Int = 0):Int {
    if (num == 1) {
        return 1 + total
    }
    return sum(num - 1, num + total)
}
```
